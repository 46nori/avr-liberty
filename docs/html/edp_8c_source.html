00413 void edpSlaveReceiveService(u08 receiveDataLength, u08* receiveData)
00414 {
00415         u08 i,checksum;
00416 
00417         // initialize the reply length from this command
00418         EdpSlaveReplyLength = 0;
00419         // verify the checksum
00420         // initialize the checksum with 1
00421         checksum = 0x01;
00422         // sum all the data in the packet and the data's checksum
00423         for(i=0; i<receiveDataLength; i++)
00424         {
00425                 checksum += receiveData[i];
00426         }
00427         // if the checksum is non-zero, then the data is corrupt
00428         if(checksum)
00429         {
00430                 // set reply code
00431                 // [FIX] which should it be?
00432                 EdpSlaveResponseCode = EDP_RESP_DATA_CHK_ERROR;
00433                 //EdpSlaveResponseCode = EDP_RESP_CMD_CHK_ERROR;
00434                 return;
00435         }
00436 
00437         // make an EDP command pointer to the received I2C data
00438         EdpCommand* edpCommand = (EdpCommand*)receiveData;
00439 
00440         // if a slave handler is defined
00441         if(edpSlaveHandlerFunc)
00442         {
00443                 // then use it
00444                 EdpSlaveResponseCode = edpSlaveHandlerFunc(
00445                         receiveDataLength, edpCommand, 
00446                         EDP_REPLY_BUFFER_SIZE, (EdpReply*)EdpSlaveReplyBuffer);
00447         }
00448         else
00449         {
00450                 // otherwise reply with unknown command
00451                 EdpSlaveResponseCode = EDP_RESP_UNKWN_CMD;
00452         }
