00055 void megaioWriteReg(unsigned char regnum, unsigned char nbytes, unsigned long data)
00056 {
00057         u08 packet[5];
00058 
00059         // construct I2c data packet
00060         //   first byte is register address
00061         //   following bytes are the data that will be written to that register
00062         packet[0] = regnum;
00063         packet[1] = data;
00064         packet[2] = data>>8;
00065         packet[3] = data>>16;
00066         packet[4] = data>>24;
00067         // send 2 bytes (register and data) to MegaIO
00068         i2cMasterSend(MEGAIO_I2C_ADDR, 1+nbytes, packet);
00069 }
