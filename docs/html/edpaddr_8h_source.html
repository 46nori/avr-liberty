00021 // The 8 bits of an EDP address byte breakdown as follows:
00022 //  b0: R/W bit (1=read, 0=write)
00023 //  b1-4: subsystem address (16 availible)
00024 //  b5-7: satellite network mask (up to 7 unique networks with 1 reserved by I2C)
00025 //  The ground station is a special case which uses 1110xxxx
00026 #define EDPNET_MASK             0xE0    // mask for satellite/ground networks
00027 #define EDPADDR_MASK            0x1E    // mask for subsystem addresses

00033 #define EDPNET_SAT_D            0x60    // EM-4
